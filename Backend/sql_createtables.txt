CREATE TABLE `comments` (   `idcomments` int(11) NOT NULL AUTO_INCREMENT,   `event_id` int(11) NOT NULL,   `user_id` int(11) NOT NULL,   `comment` varchar(150) DEFAULT NULL,   PRIMARY KEY (`idcomments`),   KEY `event_id_idx` (`event_id`),   KEY `user_id_idx` (`user_id`),   CONSTRAINT `event_id` FOREIGN KEY (`event_id`) REFERENCES `events` (`idevents`),   CONSTRAINT `user_id` FOREIGN KEY (`user_id`) REFERENCES `users` (`idusers`) ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
CREATE TABLE `event_types` (   `idevent_types` int(11) NOT NULL AUTO_INCREMENT,   `event_type` varchar(45) NOT NULL,   PRIMARY KEY (`idevent_types`) ) ENGINE=InnoDB AUTO_INCREMENT=9 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
CREATE TABLE `events` (   `idevents` int(11) NOT NULL AUTO_INCREMENT,   `name` varchar(45) NOT NULL,   `description` varchar(255) DEFAULT NULL,   `city` varchar(45) NOT NULL,   `zip` int(5) NOT NULL,   `street` varchar(45) NOT NULL,   `house_number` int(11) NOT NULL,   `address_additional` varchar(45) DEFAULT NULL,   `instagram_hashtag` varchar(45) DEFAULT NULL,   `verification_status` tinyint(4) NOT NULL,   `start_date` datetime NOT NULL,   `end_date` datetime NOT NULL,   `event_link` varchar(255) DEFAULT NULL,   `ticket_link` varchar(255) DEFAULT NULL,   PRIMARY KEY (`idevents`) ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
CREATE TABLE `saved_events` (   `idsaved_events` int(11) NOT NULL AUTO_INCREMENT,   `user_id` int(11) NOT NULL,   `event_id` int(11) NOT NULL,   PRIMARY KEY (`idsaved_events`),   KEY `_event_id_idx` (`event_id`),   KEY `_user_id_idx` (`user_id`),   CONSTRAINT `_event_id` FOREIGN KEY (`event_id`) REFERENCES `events` (`idevents`),   CONSTRAINT `_user_id` FOREIGN KEY (`user_id`) REFERENCES `users` (`idusers`) ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
CREATE TABLE `users` (   `idusers` int(11) NOT NULL AUTO_INCREMENT,   `facebook_id` varchar(45) DEFAULT NULL,   `name` varchar(45) NOT NULL,   `picture` longblob,   `email` varchar(150) DEFAULT NULL,   `password` varchar(45) DEFAULT NULL,   PRIMARY KEY (`idusers`) ) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
